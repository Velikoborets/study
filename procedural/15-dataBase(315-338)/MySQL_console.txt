// ******* ОСНОВНЫЕ КОМАНДЫ К  БД *********//

- Создание БД
CREATE DATABASE artur_test;

CREATE DATABASE my_db CHARACTER SET utf8 COLLATE utf8_general_ci;

show tables;

- Удаление БД:

// ******* РАСШИФРОВКА ЗАПРОСЫ ТАБЛИЦЫ  *********//

mysql> describe users;
+-------+--------------+------+-----+---------+----------------+
| Field | Type         | Null | Key | Default | Extra          |
+-------+--------------+------+-----+---------+----------------+
| id    | int(11)      | NO   | PRI | NULL    | auto_increment |
| email | varchar(233) | NO   | UNI | NULL    |                |
| name  | varchar(200) | NO   |     | NULL    |                |
+-------+--------------+------+-----+---------+----------------+
3 rows in set (0.01 sec)

FIELD - поле (столбец), строки наз. записями;
Type - тип данных
Null - поле не может быть null (`email` VARCHAR(255) NOT NULL)
Key - столбец, значения в котором будут всегда уникальными

PRIMARY KEY - Это указывает, что столбец id является первичным ключом таблицы. Первичный ключ - это столбец или набор столбцов, которые уникально идентифицируют каждую запись в таблице.

AUTO_INCREMENT -  указывает, что значение в этом столбце будет автоматически увеличиваться на 1 для каждой новой записи в таблиц

UNIQUE(имя столбца) - Уникальность, гарантирует, что значения в столбце email будут уникальными для каждой записи в таблице
(Например, email двух одинаковых быть не может, а вот зарплаты и имена могут, так, что )

- ENGINE=InnoDB указывает, что таблица будет использовать механизм хранения InnoDB, который обеспечивает поддержку транзакций, внешних ключей и других функций.

DEFAULT CHARSET=utf8 устанавливает кодировку по умолчанию для таблицы в UTF-8, что позволяет хранить символы Юникода.



// ******* КОМАНДЫ К СТОЛБЦАМ ТАБЛИЦ  *********//

- Добавляем новый столбец:
ALTER TABLE имя_таблицы ADD имя_столбца тип_данных

- Меняем имя столбца:
ALTER TABLE имя_таблицы RENAME COLUMN старое_имя_столбца TO новое_имя_столбца;

- Удаляем столбец:
ALTER TABLE имя_таблицы DROP имя_столбца



// ******* КОМАНДЫ К ТАБЛИЦЕ  *********//

- Структура таблицы (users - название таблицы):
